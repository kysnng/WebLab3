const h = document.getElementById('h'), m = document.getElementById('m'), s = document.getElementById('s'); function setTime() { const d = new Date(); const sec = d.getSeconds(), min = d.getMinutes(), hr = d.getHours() % 12; const sDeg = sec * 6; const mDeg = min * 6 + sec * 0.1; const hDeg = hr * 30 + min * 0.5; s.style.transform = `translate(-50%,-100%) rotate(${sDeg}deg)`; m.style.transform = `translate(-50%,-100%) rotate(${mDeg}deg)`; h.style.transform = `translate(-50%,-100%) rotate(${hDeg}deg)`; } setTime(); setInterval(setTime, 1000);
